CREATE TABLE Brukere (
    PersonID int UNIQUE NOT NULL AUTO_INCREMENT,
    Email varchar(255),
    Passord varchar(45),
    PRIMARY KEY (PersonID)
);

CREATE TABLE Bilbliotek (
    bibID int UNIQUE NOT NULL,
    bibNavn varchar(45),
    PRIMARY KEY (bibID),
    FOREIGN KEY (PersonID) REFERENCES Brukere(PersonID)
);

CREATE TABLE Mapper (
    mapID int UNIQUE NOT NULL,
    MappeNavn varchar(45),
    PRIMARY KEY (mapID),
    FOREIGN KEY (bibID) REFERENCES Bilbliotek(bibID)
);

CREATE TABLE Links (
    LinksBilde BLOB,
    LinkNavn varchar(45),
    PRIMARY KEY (mapID),
    FOREIGN KEY (mapID) REFERENCES Mapper(mapID)
);

CREATE TABLE Filer (
    filID BLOB UNIQUE NOT NULL AUTO_INCREMENT,
    PRIMARY KEY (mapID),
    FOREIGN KEY (mapID) REFERENCES Mapper(mapID)
);

CREATE TABLE Bilder (
    bildeID UNIQUE NOT NULL AUTO_INCREMENT,
    PRIMARY KEY(bibID),
    FOREIGN KEY (bibID) REFERENCES Mapper(mapID)
);
